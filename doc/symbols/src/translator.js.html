124 /**
125  * This function will sort the arguments givenen in the  for triggers/listeners in a way, that the actual devices will be able to receive them in the right order at initialisation phase.
126  * @param {string} device_type - The name of the device which trigger/listens to the function.
127  * @param {string} func_name - The name of the trigger/listener function.
128  * @param {string[]} args - The arguments to short.
129  * @param {integer} index_i - The "i" index of the loop the caller was in. The function gives this back in the callback, so that the walking of a tree can continue despite async execution.
130  * @param {integer} index_j - The "j" index of the loop the caller was in. The function gives this back in the callback, so that the walking of a tree can continue despite async execution.
131  * @param {integer} index_k - The "k" index of the loop the caller was in. The function gives this back in the callback, so that the walking of a tree can continue despite async execution.
132  * @param {integer} index_l - The "l" index of the loop the caller was in. The function gives this back in the callback, so that the walking of a tree can continue despite async execution.
133  * @param {function} callback(arg_array) - Callback function giving back the sorted argument string array.
134  */
135 sortArgs = function(device_type, func_name, args, index_i, index_j, index_k, index_l, callback){
136 	
137 	if(!args){
138 		callback([""], index_i, index_j, index_k, index_l); //Return this if no arguments were given.
139 		return;
140 	}
141 	
142 	var _db = new sqlite3.Database("/home/mogyula/devel/dabox/id.db",sqlite3.OPEN_READONLY); //Open our database.
143 			
144 	_db.all("SELECT"+
145 	" type_id.name AS type_name,"+
146 	" func_id.arg_cnt AS arg_cnt,"+
147 	" func_id.name AS func_name,"+
148 	" arg_order.arg_no AS arg_no,"+
149 	" arg_order.arg_name AS arg_name"+
150 	" FROM type_id INNER JOIN func_id"+
151 	" ON type_id.type_id=func_id.type_id"+
152 	" INNER JOIN arg_order"+
153 	" ON type_id.type_id=arg_order.type_id"+
154 	" AND func_id.func_id=arg_order.func_id"+
155 	" WHERE type_name='"+device_type+"' AND"+
156 	" func_name='"+func_name+"'"
157 	,function(err, row){
158 		
159 		if(err){
160 			callback(null, null, null, null, null, err);
161 			return;
162 		}
163 		
164 		//TODO: error handling
165 		//create an array with length arg_cnt, if it doesn't exist yet
166 		var	_arg_array = new Array(row[0].arg_cnt);
167 		
168 		//Let's search for that attribute amongst the given ones...
169 		for (var i in args){
170 			for (var j in row){
171 				if (args[i].name==row[j].arg_name){
172 					//...and put it in this array in order.
173 					_arg_array[--row[j].arg_no] = args[i].$t;
174 				}
175 			}
176 		}
177 		callback(_arg_array, index_i, index_j, index_k, index_l, null);
178 		return;
179 	});
180 	_db.close();
181 	return;
182 }

184 getDeviceFunctionID = function(deviceName, functionName, callback){
185 	var _db = new sqlite3.Database("/home/mogyula/devel/dabox/id.db",sqlite3.OPEN_READONLY);
186 	
187 	_db.each("SELECT type_id.type_id AS type_id, func_id.func_id AS func_id"+
188 	" FROM type_id INNER JOIN func_id"+
189 	" ON type_id.type_id=func_id.type_id"+
190 	" WHERE type_id.name='"+deviceName+"'"+
191 	" AND func_id.name='"+functionName+"'"
192 	,function(err, row){
193 		
194 		if (err){
195 			callback(null, null, err);
196 			return;
197 		}
198 		
199 		callback(row.type_id, row.func_id, null);
200 		return;
201 	});
202 	_db.close();
203 	return;

206 /**
207  * This function counts the <listener> elements in the XML tree, so that the total number of trigger-listener pairs can be known.
208  * @param {object} obj - The JS object containing the parsed XMl tree
209  */
210 getListenerCount = function(obj){
211 	var _listenerCount=0;
212 	
213 	for (var i in obj.system[0].device){
214 		for (var j in obj.system[0].device[i].trigger){
215 			for (var k in obj.system[0].device[i].trigger[j].device){
216 					_listenerCount+=obj.system[0].device[i].trigger[j].device[k].listener.length;
217 			}
218 		}
219 	}
220 	return _listenerCount;

223 /**
224  * This function checks the semantics of the JS object parsed from the initial XML
225  * @param {object} obj - The JS object containing the parsed XML tree
226  */
227 checkSyntax = function(obj){
228 	var errorMsg="Error while translating XML: " //every error message will start with this
229 	var children; //just a loop counter
230 
231 //TODO: check everywhere, that the given number of args were given
232 	
233 	//only one <system> can and should exist.
234 	if (Object.keys(obj).length!=1 || Object.keys(obj)[0]!="system"){
235 		throw errorMsg+"only a single <system>...</system> element should exist at the top level!\n   "+
236 		"In: system";
237 	}
238 	
239 	//only <device> element(s) should be in the <system>
240 	var system_keys = Object.keys(obj.system[0]);
241 	for(children in system_keys){
242 		if (system_keys[children]!="device"){
243 			throw errorMsg+"the <system> element should only contain one or more <device> elements!\n   "+
244 			"In: system";
245 		}
246 	}
247 	
248 	//iterating outer <device> elements...
249 	for (var i in obj.system[0].device){
250 		
251 		//only <device type=... id=...> should exist.
252 		var outer_device_attributes=getAttributes(obj.system[0].device[i]);
253 		if (outer_device_attributes.length!=2 ||
254 			outer_device_attributes.indexOf("type") == -1 ||
255 			outer_device_attributes.indexOf("id") == -1){
256 			throw errorMsg+"the <device> elements must have one \"type\", and one \"id\" property defined!\n   "+
257 			"In: system > device("+obj.system[0].device[i].type+")\n";
258 		}
259 		
260 		//Members of outer <device>s must be <trigger>s
261 		var outer_device_keys = Object.keys(obj.system[0].device[i]);
262 		for(children=2; children<outer_device_keys.length; children++){ //the 0 and 1 index was the "type" and "id" attributes.
263 			if (outer_device_keys[children]!="trigger"){
264 				throw errorMsg+"the outer <device> element should only contain one or more <trigger> elements!\n\   "+
265 				"In: system > device("+obj.system[0].device[i].type+")\n";
266 			}
267 		}
268 		
269 		//iterating through <trigger>s...
270 		for (var j in obj.system[0].device[i].trigger){
271 			
272 			var trigger_attributes = getAttributes(obj.system[0].device[i].trigger[j]);
273 			if (trigger_attributes.length != 1 ||
274 				trigger_attributes.indexOf("name") == -1){
275 				throw errorMsg+"the <trigger> element should have one \"name\" property defined!\n   "+
276 				"In: system > device("+obj.system[0].device[i].type+")\n";
277 			}
278 			
279 			//iterating through trigger <arg>s and <device>s inside <trigger>s
280 			var trigger_keys = Object.keys(obj.system[0].device[i].trigger[j]);
281 			for(children=1 ;children<trigger_keys.length;children++){
282 				
283 				//in case no device to listen was given
284 				if(trigger_keys.indexOf("device")==0){
285 					throw errorMsg+"the <trigger> element should contain at least one <device> element!\n   "+
286 					"In: system > device("+obj.system[0].device[i].type+") > "+
287 					"trigger("+obj.system[0].device[i].trigger[j].name+")\n";
288 				}
289 				
290 				//in case anything else than <arg> or <device> was inside the <trigger> element
291 				if (trigger_keys[children]!="arg" &&
292 					trigger_keys[children]!="device"){
293 					throw errorMsg+"the <trigger> element should only contain <arg> and <device> elements!\n   "+
294 					"In: system > device("+obj.system[0].device[i].type+") > "+
295 					"trigger("+obj.system[0].device[i].trigger[j].name+")\n";
296 				}
297 			}
298 			
299 			//looping through the <arg> elements to see if they only have a name attribute and and a value inside
300 			for (children in obj.system[0].device[i].trigger[j].arg){
301 				
302 				var trigger_arg_attributes=getAttributes(obj.system[0].device[i].trigger[j].arg[children]);
303 				//args should have a name
304 				if (trigger_arg_attributes[0]!="name"){
305 					throw errorMsg+"the <arg> elements must have a \"name\" property defined!\n   "+
306 					"In: system > device("+obj.system[0].device[i].type+") > "+
307 					"trigger("+obj.system[0].device[i].trigger[j].name+")\n";
308 				}
309 				//args should have a value
310 				if (trigger_arg_attributes[1]!="$t"){
311 					throw errorMsg+"the <arg> elements must have a defined value!\n   "+
312 					"In: system > device("+obj.system[0].device[i].type+") > "+
313 					"trigger("+obj.system[0].device[i].trigger[j].name+") > "+
314 					"arg("+obj.system[0].device[i].trigger[j].arg[children].name+")\n";
315 				}
316 			}
317 			
318 			//looping through the inner <device>s
319 			for (var k in obj.system[0].device[i].trigger[j].device){
320 				
321 				//check if only the "type" and "id" attributes were given to the inner <device> element
322 				var inner_device_attributes = getAttributes(obj.system[0].device[i].trigger[j].device[k]);
323 				if (inner_device_attributes.length != 2 ||
324 					inner_device_attributes.indexOf("type")==-1 ||
325 					inner_device_attributes.indexOf("id")==-1){
326 					throw errorMsg+"the <device> elements must have one \"type\", and one \"id\" property defined!\n   "+
327 					"In: system > device("+obj.system[0].device[i].type+") > "+
328 					"trigger("+obj.system[0].device[i].trigger[j].name+")\n"
329 				}
330 				
331 				var inner_device_keys=Object.keys(obj.system[0].device[i].trigger[j].device[k]);
332 				for(children=2; children<inner_device_keys.length;children++){
333 					if(inner_device_keys[children]!="listener"){
334 						throw errorMsg+"the inner <device> element should only contain <listener> elements\n   "+
335 						"In: system > device("+obj.system[0].device[i].type+") > "+
336 						"trigger("+obj.system[0].device[i].trigger[j].name+") > "+
337 						"device("+obj.system[0].device[i].trigger[j].device[k].type+")\n";
338 					}
339 				}
340 				
341 				//looping through the <listener> elements
342 				for (var l in obj.system[0].device[i].trigger[j].device[k].listener){
343 					
344 					//checking that the listener element has only one attribute, which is "name"
345 					var listener_attributes=getAttributes(obj.system[0].device[i].trigger[j].device[k].listener[l]);
346 					if (listener_attributes.length!=1 ||
347 						listener_attributes.indexOf("name")==-1){
348 						throw "the <listener> element should have one \"name\" property defined!\n   "+
349 						"In: system > device("+obj.system[0].device[i].type+") > "+
350 						"trigger("+obj.system[0].device[i].trigger[j].name+") > "+
351 						"device("+obj.system[0].device[i].trigger[j].device[k].type+") > "+
352 						"listener("+obj.system[0].device[i].trigger[j].device[k].listener[l].name+")\n";
353 					}
354 					
355 					//checking that only <arg>s are in <listener>s
356 					var inner_device_keys=Object.keys(obj.system[0].device[i].trigger[j].device[k].listener[l]);
357 					for(children=1; children<inner_device_keys.length; children++){
358 						if(inner_device_keys[children]!="arg"){
359 							throw "the <listener> element should only contain <arg> elements!\n   "+
360 							"In: system > device("+obj.system[0].device[i].type+") > "+
361 							"trigger("+obj.system[0].device[i].trigger[j].name+") > "+
362 							"device("+obj.system[0].device[i].trigger[j].device[k].type+") > "+
363 							"listener("+obj.system[0].device[i].trigger[j].device[k].listener[l].name+")\n";
364 						}
365 					}
366 					
367 					//looping through <arg>s of listeners
368 					for (children in obj.system[0].device[i].trigger[j].device[k].listener[l].arg){
369 						
370 						var listener_arg_attributes = getAttributes(obj.system[0].device[i].trigger[j].device[k].listener[l].arg[children]);
371 						
372 						//checking if they all have a "name" attribute
373 						if (listener_arg_attributes[0]!="name"){
374 							throw errorMsg+"the <arg> elements must have a \"name\" property defined!\n   "+
375 							"In: system > device("+obj.system[0].device[i].type+") > "+
376 							"trigger("+obj.system[0].device[i].trigger[j].name+") > "+
377 							"device("+obj.system[0].device[i].trigger[j].device[k].type+") > "+
378 							"listener("+obj.system[0].device[i].trigger[j].device[k].listener[l].name+")\n";
379 						}
380 				
381 						//and a value
382 						if (listener_arg_attributes[1]!="$t"){
383 							throw errorMsg+"the <arg> elements must have a defined value!\n   "+
384 							"system > device("+obj.system[0].device[i].type+") > "+
385 							"trigger("+obj.system[0].device[i].trigger[j].name+") > "+
386 							"device("+obj.system[0].device[i].trigger[j].device[k].type+") > "+
387 							"listener("+obj.system[0].device[i].trigger[j].device[k].listener[l].name+") > "+
388 							"arg("+obj.system[0].device[i].trigger[j].device[k].listener[l].arg[children].name+")\n";
389 						}
390 					}
391 				}
392 			}
393 		}
394 	}
395 	return; //if there was no error

398 /**
399  * This function returns an array of those top-level elements that have no children in the object.
400  * This corresponds to the attributes and values of elements (not sub-elements) of the XML tree.
401  * @param {object} obj - A JS object
402  */
403 getAttributes = function(obj){
404 	var result = [];
405 
406 	for (var label in obj){
407 		if(typeof(obj[label])=="string"){
408 			result.push(label);
409 		}
410 	}
411 	return result;
